#include<bits/stdc++.h>
using namespace std;

class Student{

public:
    string name;
    int roll;
    string branch;
    int cgpa;
    int *arr = new int[5];

    Student()
    {
    name = "Tell your name";
    roll = 00;
    branch = "Which branch?";
    }

    Student(string s , int r , string b) 
    {
    name = s;
    roll = r;
    branch = b;
    cgpa = 10;
    }

    void totalmarks()
    {
    int sum = 0;
    cout<<"Enter Marks"<<endl;
    for(int i=0; i<5 ; i++){
        cin >> arr[i];
        sum = sum + arr[i];
    }
    cout << "Marks: " << sum << endl;
}

    void Cgpa() 
    {
    int sum = 0;
    cout<<"Enter Marks"<<endl;
    for(int i=0;i<5;i++){
        cin >> arr[i];
        sum = sum + arr[i];
    }
    int percentage = (sum / 500)*100;
    cgpa = percentage/9.5;
    cout << "CGPA IS : " << cgpa << endl;
    }
    void getInput()
    {
        cout << "Give the input:\n";
        cout << "Name: ";
        cin >> name;
        cout << "Roll No: ";
        cin >> roll;
        cout << "Branch: ";
        cin >> branch;
        cout << "Marks: ";
        cin >> arr[0];
        cin >> arr[1];
        cin >> arr[2];
        cin >> arr[3];
        cin >> arr[4];
        
    }

    void displayOutput() {
    
        cout << "Name: " << name << "\n";
        cout << "Roll No:" << roll << endl;
        cout << "Branch: " << branch << endl;
        cout << "CGPa: " << cgpa << endl;
    }

};
int main() {
    Student s("Juhi",60,"IT");
    Student p;
    s.displayOutput();
    s.totalmarks();
    p.getInput();
}
